  <script>
    // Diseño del corazón en una cuadrícula 12x10
    const heartPattern = [
      "00111100011100",
      "01111110111110",
      "11111111111111",
      "11111111111111",
      "01111111111110",
      "00111111111100",
      "00011111111000",
      "00001111110000",
      "00000111100000",
      "00000011000000"
    ];

    const container = document.getElementById("heartContainer");
    let delay = 0;

    heartPattern.forEach((row, r) => {
      row.split("").forEach((cell, c) => {
        if (cell === "1") {
          const block = document.createElement("div");
          block.className = "heart";
          block.style.gridRowStart = r + 1;
          block.style.gridColumnStart = c + 1;
          block.style.animationDelay = `${delay * 0.15}s`;
          container.appendChild(block);
          delay++;
        }
      });
    });
  </script>
